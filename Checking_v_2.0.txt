#include <iostream>
#include <string>
using namespace std;

int checker_int(string text, int min, int max)
{
    string str;
    bool state = true;
    bool state_1 = true;
    bool state_2 = true;
    int count = 0;
    int minus = 0;
    int num = 0;
    cout << text << endl;
    while (state) {
        getline(cin, str);
        if (str.empty()) {
            cout << "The input is empty. Enter a number:" << endl;
            state_1 = false;
            state_2 = false;           
        }
        else {
            state_1 = true;
            state_2 = true;
            count = 0;
            minus = 0;
            while (count < str.length() and state_1) {
                if ((str[count] == '-')) {
                    minus += 1;
                }
                if (!isdigit(str[count]) and str[0] != '-' or (minus>1) or (str[count] ==  '-' and !isdigit(str[count + 1]))) {
                    cout << "The input contains invalid characters. Enter a correct input:" << endl;
                    state_1 = false;
                    state_2 = false;
                }
                count++;
            }
        }
        try {
            num = stoi(str);
        }
        catch (...) { state_2 = 0; }
        if ((num < min or num > max) and state_2) {
            cout << "The input is not included in the boundaries. Enter a correct input:" << endl;
            state_1 = false;
            state_2 = false;
        }
        if (state_1)
            state = 0;
    }
    return num;
}
